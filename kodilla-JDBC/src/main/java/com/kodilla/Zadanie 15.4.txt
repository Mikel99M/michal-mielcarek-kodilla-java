CREATE TABLE TASKS
(
    ID SERIAL PRIMARY KEY,
    USER_ID BIGINT UNSIGNED NOT NULL,
    TASK BIGINT UNSIGNED NOT NULL,
    FOREIGN KEY (USER_ID) REFERENCES USERS(ID),
    FOREIGN KEY (TASK) REFERENCES ISSUES(ID)
);

SELECT * FROM TASKS;

SELECT T.USER_ID, U.FIRSTNAME, U.LASTNAME, I.SUMMARY AS TASKS_ASSIGNED
FROM TASKS T
    JOIN USERS U ON T.USER_ID = U.ID
    JOIN ISSUES I ON T.TASK = I.ID;

SELECT U.FIRSTNAME, U.LASTNAME, COUNT(*) AS TASKS_NUMBER
FROM TASKS T JOIN USERS U ON T.USER_ID = U.ID
GROUP BY FIRSTNAME, LASTNAME;
